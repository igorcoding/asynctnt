    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """
        Normally `commit()` is called on context manager exit, but
        in the case of exception `rollback()` is called
        """
        if exc_type and exc_val:
            await self.rollback()
        else:
            await self.commit()
