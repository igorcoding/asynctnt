[docs]class TarantoolDatabaseError(TarantoolError):
    """
        Exception is raised when Tarantool responds with code != 0
    """
    def __init__(self,
                 code: int,
                 message: str,
                 error: Optional[IProtoError]):
        super(TarantoolDatabaseError, self).__init__(message)
        self.code = code
        self.message = message
        self.error = error
